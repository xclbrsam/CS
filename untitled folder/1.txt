2. View Operations

USE dbms;

CREATE TABLE stud (
    RollNo INT,
    Name VARCHAR(20),
    Class VARCHAR(20)
);

INSERT INTO stud VALUES (1, 'ABC', 'comp');
INSERT INTO stud VALUES (2, 'qaz', 'comp');
INSERT INTO stud VALUES (3, 'wsx', 'comp');

SELECT * FROM stud;

CREATE VIEW std AS
SELECT RollNo, Name FROM stud;

SELECT * FROM std;

CREATE OR REPLACE VIEW std AS
SELECT Name FROM stud WHERE Name = 'wsx';

SELECT * FROM std;

DROP VIEW std;

SELECT * FROM std;
